//question is check if string is substring or sequence of other //i/p s1= "abc" s2="ahbgdc" m = 3 , n=6 function isSubSequence(s1,s2,m,n){ if(n === 0){ return true; //s2 is an empty string } if(m === 0){ return false; //s1 is an empty string } if(s1[m-1] === s2[n-1]) //if last characters of both strings match, call again with new lengths, remove last charcaters return isSubSequence(s1,s2,m-1,n-1); else //if not , remove last from s1 but keep s2 return isSubSequence(s1,s2,m-1,n); } //question is check if string is anagram //s1=geeks s2=skeeg
